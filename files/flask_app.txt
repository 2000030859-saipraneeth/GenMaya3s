#src/app.py
from flask import Flask, render_template, request, jsonify
import math
import operator
import re

app = Flask(__name__)

@app.route('/', methods=['GET', 'POST'])
def main():
    if request.method == 'POST':
        try:
            user_input = request.form['user_input']
            if user_input == '':
                return jsonify({'error': 'Please enter a mathematical expression.'})
            elif user_input =='sin(x)' or user_input == 'cos(x)' or user_input == 'tan(x)' or user_input == 'log(x)':
                return jsonify({'error': 'Invalid input. Please enter a mathematical expression.'})
            elif user_input =='sqrt(x)' or user_input == 'log(x)':
                return jsonify({'error': 'Invalid input. Please enter a mathematical expression.'})
            else:
                try:
                    result = eval(user_input)
                    return jsonify({'result': result})
                except ZeroDivisionError:
                    return jsonify({'error': 'Division by zero is not allowed.'})
                except Exception as e:
                    return jsonify({'error': str(e)})
        except Exception as e:
            return jsonify({'error': str(e)})

    return render_template('index.html')

if __name__ == '__main__':
    app.run(debug=True)
